\section{Preparing Eclipse}

\subsection{Preparing Eclipse 1/4}
\begin{frame}[fragile]
	\frametitle{\insertsubsection}
	\begin{enumerate}
    \item Download Eclipse: \url{http://www.eclipse.org/downloads/}
		\begin{itemize}
			\item 3.6 (Helios) is recommend (should also work with 3.4 and 3.5)
			\item Choose "Eclipse for RCP and RAP Developers" that you can access sources of Eclipse standard plug-ins
		\end{itemize}
		\item Unzip Eclipse
		\begin{itemize}
			\item Make sure that you have all permissions for the directory (do not use Windows' program files folder)
		\end{itemize}
		\item Create a shortcut, add VM parameters: \verb|.../eclipse.exe|
			\verb|-vmargs -Duser.name="Name Surname" -Xmx1024M|
		\begin{itemize}
			\item Eclipse can automatically insert your name as author
			\item Avoids OutOfMemoryException
		\end{itemize}
		\item Start Eclipse and create a new workspace
    \setcounter{saveenum}{\value{enumi}}
	\end{enumerate}
\end{frame}
